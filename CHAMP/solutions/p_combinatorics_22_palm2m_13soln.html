&= \frac{1}{|S|} \sum_{A \in S} g(A) \\
&= \frac{1}{|S|} \sum_{A \in S} \min(g(A)) \\
&= \frac{1}{|S|} \sum_{A \in S} \min(A) \\
&= x.
\end{align*}
